Chord Chart

import {
  Chord,
  useWidth,
} from 'cl-react-graph;


const MyComponent = () => {
  const [ref, width] = useWidth('90%');


  return (
    <div ref={ref}>
      <Chord
        width={width}
        height={400}
        data={
          {
            'France': [0, 10, 20, 12],
            'Britain': [20, 0, 30, 2],
            'Ireland': [30, 40, 0, 23],
            'Spain': [10, 23, 43, 0]
          }
        } />
    </div>
  )
};
